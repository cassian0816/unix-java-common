// rpc接口方法扩展设置
message RpcMethodExtOpt {
  bool is_write_operation = 1; // 是否为写操作，默认为读操作
  bool is_log_detail = 2; // 是否需要日志记录请求/返回详细信息
}

// rpc方法扩展选项
extend google.protobuf.MethodOptions {
  optional RpcMethodExtOpt rpc_method_ext_opt = 12001;
}

// 详细错误信息
// 使用2层错误码设计
//    第一层：grpc内置错误码 or http错误码 (这两者有固定转换关系)
//    第二层：业务服务自定义的错误码。为了方便前后兼容，不要直接使用enum定义。
message ErrorDetail {
  string code = 1; // 业务错误码，每个业务使用错误定义时，注意要使用固定的服务前缀.例如：TRADE_INVALID_ACCOUNT_ID
  map<string, string> param = 2; // 错误信息中的param，用于多语言展示替换, 例如: { "accountId": "-1" }
}
